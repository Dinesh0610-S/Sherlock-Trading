+// ── ATR calculator ────────────────────────────────────────────────────────────
+// Uses last N candles to compute Average True Range.
+// Falls back to 1% of spot (or symbol default) when candles unavailable.
+function calculateATR(candles, symbol = 'NIFTY', period = 14) {
+  // Fallback ATR when market is closed / no candles
+  const fallbackAtr = (sym) => {
+    if (sym === 'BANKNIFTY') return 150;
+    if (sym === 'NIFTY')     return 75;
+    return 80;
   };
+
+  if (!Array.isArray(candles) || candles.length < 2) return fallbackAtr(symbol);
+
+  const trs = [];
+  for (let i = 1; i < candles.length; i++) {
+    const curr = candles[i];
+    const prev = candles[i - 1];
+    const high = curr.high ?? curr.h ?? 0;
+    const low  = curr.low  ?? curr.l ?? 0;
+    const prevClose = prev.close ?? prev.c ?? 0;
+    if (!high || !low || !prevClose) continue;
+    const tr = Math.max(high - low, Math.abs(high - prevClose), Math.abs(low - prevClose));
+    trs.push(tr);
+  }
+  if (trs.length === 0) return fallbackAtr(symbol);
+  const slice = trs.slice(-period);
+  const avg = slice.reduce((s, v) => s + v, 0) / slice.length;
+  return Math.max(Math.round(avg), fallbackAtr(symbol) / 3); // never trivially small
 }
