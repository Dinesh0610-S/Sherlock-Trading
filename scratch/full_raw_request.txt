// TradeGate.js — enforced across all pages

class OneTradePerdayGate {
  constructor() {
    this.todaysTrade = null;
    this.date = null;
  }

  canTrade() {
    const today = new Date().toDateString();
    if (this.date !== today) {
      this.todaysTrade = null;
      this.date = today;
    }
    return this.todaysTrade === null;
  }

  recordTrade(trade) {
    this.todaysTrade = { ...trade, recordedAt: new Date() };
    // Store in localStorage for persistence
    localStorage.setItem('todaysTrade', JSON.stringify(this.todaysTrade));
  }

  getTodaysTrade() {
    return this.todaysTrade;
  }
}

// Show on every page when trade exists:
{!gate.canTrade() && (
  <div className="trade-taken-banner">

    <span>Entry: ₹{gate.getTodaysTrade()?.entry}</span>
    <span>Status: {gate.getTodaysTrade()?.status}</span>
